132 re_status_t re_log_write_data (uint8_t * const buffer, const re_float data,
133  const uint8_t source)
134 {
135  re_status_t err_code = RE_SUCCESS;
136  int32_t discrete_value = 0xFFFFFFFF;
137  re_float scaled_value = 0;
138 
139  if (NULL == buffer)
140  {
141  err_code |= RE_ERROR_NULL;
142  }
143  else if (isnan (data) || isinf (data))
144  {
145  err_code |= RE_ERROR_INVALID_PARAM;
146  }
147  else
148  {
149  switch (source)
150  {
151  case RE_STANDARD_DESTINATION_ACCELERATION_X:
152  case RE_STANDARD_DESTINATION_ACCELERATION_Y:
153  case RE_STANDARD_DESTINATION_ACCELERATION_Z:
154  scaled_value = (data * RE_STANDARD_ACCELERATION_SF);
155  break;
156 
157  case RE_STANDARD_DESTINATION_GYRATION_X:
158  case RE_STANDARD_DESTINATION_GYRATION_Y:
159  case RE_STANDARD_DESTINATION_GYRATION_Z:
160  scaled_value = (data * RE_STANDARD_GYRATION_SF);
161  break;
162 
163  case RE_STANDARD_DESTINATION_HUMIDITY:
164  scaled_value = (data * RE_STANDARD_HUMIDITY_SF);
165  break;
166 
167  case RE_STANDARD_DESTINATION_PRESSURE:
168  scaled_value = (data * RE_STANDARD_PRESSURE_SF);
169  break;
170 
171  case RE_STANDARD_DESTINATION_TEMPERATURE:
172  scaled_value = (data * RE_STANDARD_TEMPERATURE_SF);
173  break;
174 
175  default:
176  err_code |= RE_ERROR_NOT_IMPLEMENTED;
177  break;
178  }
179 
180  scaled_value = roundf (scaled_value);
181  discrete_value = f2i (scaled_value);
182  }

184  // These shifts do not rely on the value of leftmost bit if original
185  // value is negative, so this is safe way to encode bytes.
186  buffer[RE_LOG_WRITE_VALUE_MSB_IDX] = (uint8_t) ( (discrete_value >> 24U) & 0xFFU);
187  buffer[RE_LOG_WRITE_VALUE_B2_IDX] = (uint8_t) ( (discrete_value >> 16U) & 0xFFU);
188  buffer[RE_LOG_WRITE_VALUE_B3_IDX] = (uint8_t) ( (discrete_value >> 8U) & 0xFFU);
189  buffer[RE_LOG_WRITE_VALUE_LSB_IDX] = (uint8_t) (discrete_value & 0xFFU);
