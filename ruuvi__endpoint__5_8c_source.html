24  const ruuvi_endpoint_5_data_t * data)
25 {
26  if (NULL == buffer || NULL == data) { return RUUVI_ENDPOINT_ERROR_NULL; }
27 
28  buffer[RUUVI_ENDPOINT_5_OFFSET_HEADER] = RUUVI_ENDPOINT_5_DESTINATION;
29  uint16_t humidity = RUUVI_ENDPOINT_5_INVALID_HUMIDITY;
30 
31  if (!isnan (data->humidity_rh) && 0 < data->humidity_rh)
32  {
33  // Humidity (16bit unsigned) in 0.0025% (0-163.83% range, though realistically 0-100%)
34  humidity = (uint16_t) (data->humidity_rh * 400);
35  }
36 
37  buffer[RUUVI_ENDPOINT_5_OFFSET_HUMIDITY_MSB] = (humidity >> 8);
38  buffer[RUUVI_ENDPOINT_5_OFFSET_HUMIDITY_LSB] = humidity & 0xFF;
39  // Temperature is in 0.005 degrees
40  int16_t temperature = RUUVI_ENDPOINT_5_INVALID_TEMPERATURE;
41 
42  if (!isnan (data->temperature_c))
43  {
44  //
45  temperature = (int16_t) (data->temperature_c * 200);
46  }
47 
48  buffer[RUUVI_ENDPOINT_5_OFFSET_TEMPERATURE_MSB] = (temperature >> 8);
49  buffer[RUUVI_ENDPOINT_5_OFFSET_TEMPERATURE_LSB] = (temperature & 0xFF);
50  // Pressure
51  uint32_t pressure = RUUVI_ENDPOINT_5_INVALID_PRESSURE;
52 
53  if (!isnan (data->pressure_pa) && 50000 < data->pressure_pa)
54  {
55  pressure = data->pressure_pa;
56  pressure -= 50000;
57  }
58 
59  buffer[RUUVI_ENDPOINT_5_OFFSET_PRESSURE_MSB] = (pressure >> 8) & 0xFF;
60  buffer[RUUVI_ENDPOINT_5_OFFSET_PRESSURE_LSB] = (pressure & 0xFF);
61  // acceleration
62  ruuvi_endpoint_5_encode_acceleration (&buffer[RUUVI_ENDPOINT_5_OFFSET_ACCELERATIONX_MSB],
63  data->accelerationx_g);
64  ruuvi_endpoint_5_encode_acceleration (&buffer[RUUVI_ENDPOINT_5_OFFSET_ACCELERATIONY_MSB],
65  data->accelerationy_g);
66  ruuvi_endpoint_5_encode_acceleration (&buffer[RUUVI_ENDPOINT_5_OFFSET_ACCELERATIONZ_MSB],
67  data->accelerationz_g);
68  // voltage, tx power
69  uint16_t voltage = RUUVI_ENDPOINT_5_INVALID_VOLTAGE;
70  int8_t tx_power = RUUVI_ENDPOINT_5_INVALID_POWER;
71 
72  if (!isnan (data->battery_v) && 1.6 < data->battery_v)
73  {
74  // first 11 bits unsigned is the battery voltage above 1.6V, in millivolts (1.6V to 3.647V range)
75  voltage = (data->battery_v * 1000 - 1600);
76  }
77 
78  if (RUUVI_ENDPOINT_5_INVALID_POWER != data->tx_power && data->tx_power >= -40
79  && data->tx_power <= 24)
80  {
81  // Last 5 bits unsigned is the TX power above -40dBm, in 2dBm steps. (-40dBm to +24dBm range)
82  tx_power = (data->tx_power + 40) / 2;
83  }
84 
85  uint16_t power_info = (voltage << 5) + tx_power;
86  buffer[RUUVI_ENDPOINT_5_OFFSET_POWER_MSB] = (power_info >> 8);
87  buffer[RUUVI_ENDPOINT_5_OFFSET_POWER_LSB] = (power_info & 0xFF);
88  buffer[RUUVI_ENDPOINT_5_OFFSET_MOVEMENT_COUNTER] = data->movement_count;
89  buffer[RUUVI_ENDPOINT_5_OFFSET_SEQUENCE_COUNTER_MSB] = (data->measurement_count >> 8);
90  buffer[RUUVI_ENDPOINT_5_OFFSET_SEQUENCE_COUNTER_LSB] = (data->measurement_count & 0xFF);
91  // Address is 64 bits, skip 2 first bytes
92  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB] = (data->address >> 40) & 0xFF;
93  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB + 1] = (data->address >> 32) & 0xFF;
94  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB + 2] = (data->address >> 24) & 0xFF;
95  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB + 3] = (data->address >> 16) & 0xFF;
96  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB + 4] = (data->address >> 8) & 0xFF;
97  buffer[RUUVI_ENDPOINT_5_OFFSET_ADDRESS_MSB + 5] = (data->address >> 0) & 0xFF;
98  return RUUVI_ENDPOINT_SUCCESS;
99 }
