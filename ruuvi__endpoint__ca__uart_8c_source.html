127 static re_status_t re_ca_uart_decode_adv_rprt (const uint8_t * const buffer,
128  re_ca_uart_payload_t * const payload)
129 {
130  re_status_t err_code = RE_SUCCESS;
131  const uint8_t adv_len = buffer[RE_CA_UART_LEN_INDEX]
132  - RE_CA_UART_MAC_BYTES
133  - RE_CA_UART_RSSI_BYTES
134  - (RE_CA_UART_MAXFIELDS) * RE_CA_UART_DELIMITER_LEN;
135  const uint8_t * const p_rssi = buffer
136  + RE_CA_UART_PAYLOAD_INDEX
137  + buffer[RE_CA_UART_LEN_INDEX]
138  - RE_CA_UART_RSSI_BYTES
139  - RE_CA_UART_DELIMITER_LEN;
140  const uint8_t * const p_data = buffer
141  + RE_CA_UART_PAYLOAD_INDEX
142  + RE_CA_UART_MAC_BYTES
143  + RE_CA_UART_DELIMITER_LEN;
144 
145  if (adv_len > RE_CA_UART_ADV_BYTES)
146  {
147  err_code |= RE_ERROR_DECODING;
148  }
149  else if (RE_CA_UART_FIELD_DELIMITER != * (p_data - RE_CA_UART_DELIMITER_LEN))
150  {
151  err_code |= RE_ERROR_DECODING;
152  }
153  else if (RE_CA_UART_FIELD_DELIMITER != * (p_rssi - RE_CA_UART_DELIMITER_LEN))
154  {
155  err_code |= RE_ERROR_DECODING;
156  }
157  else if (RE_CA_UART_FIELD_DELIMITER != * (p_rssi + RE_CA_UART_RSSI_BYTES))
158  {
159  err_code |= RE_ERROR_DECODING;
160  }
161  else
162  {
163  memcpy (payload->params.adv.mac,
164  buffer + RE_CA_UART_PAYLOAD_INDEX,
165  RE_CA_UART_MAC_BYTES);
166  memcpy (payload->params.adv.adv,
167  p_data,
168  adv_len);
169  payload->cmd = RE_CA_UART_ADV_RPRT;
170  payload->params.adv.rssi_db = u8toi8 (*p_rssi);
171  payload->params.adv.adv_len = adv_len;
172  }
173 
174  return err_code;

177 re_status_t re_ca_uart_decode (const uint8_t * const buffer,
178  re_ca_uart_payload_t * const payload)
179 {
180  re_status_t err_code = RE_SUCCESS;
181 
182  // Sanity check buffer format
183  if (NULL == buffer)
184  {
185  err_code |= RE_ERROR_NULL;
186  }
187  else if (NULL == payload)
188  {
189  err_code |= RE_ERROR_NULL;
190  }
191  else if (RE_CA_UART_STX != buffer[RE_CA_UART_STX_INDEX])
192  {
193  err_code |= RE_ERROR_DECODING;
194  }
195  else if (RE_CA_UART_ETX != buffer[buffer[RE_CA_UART_LEN_INDEX] + RE_CA_UART_HEADER_SIZE])
196  {
197  err_code |= RE_ERROR_DECODING;
198  }
199  else if (RE_CA_UART_NOT_CODED != payload->cmd)
200  {
201  err_code |= RE_ERROR_INVALID_PARAM;
202  }
203  else
204  {
205  switch (buffer[RE_CA_UART_CMD_INDEX])
206  {
207  case RE_CA_UART_SET_FLTR:
208  err_code |= re_ca_uart_decode_set_fltr (buffer, payload);
209  break;
210 
211  case RE_CA_UART_CLR_FLTR:
212  err_code |= re_ca_uart_decode_clr_fltr (buffer, payload);
213  break;
214 
215  case RE_CA_UART_SET_CH:
216  err_code |= re_ca_uart_decode_set_ch (buffer, payload);
217  break;
218 
219  case RE_CA_UART_SET_PHY:
220  err_code |= re_ca_uart_decode_set_phy (buffer, payload);
221  break;
222 
223  case RE_CA_UART_ADV_RPRT:
224  err_code |= re_ca_uart_decode_adv_rprt (buffer, payload);
225  break;
226

228  err_code |= RE_ERROR_DECODING;

231  }

236 static re_status_t re_ca_uart_encode_adv_rprt (uint8_t * const buffer,
237  uint8_t * const buf_len,
238  const re_ca_uart_payload_t * const payload)
239 {
240  re_status_t err_code = RE_SUCCESS;
241  uint8_t written = 0;
242 
243  if (RE_CA_UART_TX_MAX_LEN > *buf_len)
244  {
245  err_code |= RE_ERROR_DATA_SIZE;
246  }
247  else
248  {
249  buffer[RE_CA_UART_STX_INDEX] = RE_CA_UART_STX;
250  // Payload length is different from total message length.
251  buffer[RE_CA_UART_LEN_INDEX] = RE_CA_UART_MAC_BYTES
252  + payload->params.adv.adv_len
253  + RE_CA_UART_RSSI_BYTES
254  + RE_CA_UART_ADV_FIELDS * RE_CA_UART_DELIMITER_LEN;
255  buffer[RE_CA_UART_CMD_INDEX] = RE_CA_UART_ADV_RPRT;
256  written += RE_CA_UART_HEADER_SIZE;
257  memcpy (buffer + RE_CA_UART_PAYLOAD_INDEX,
258  payload->params.adv.mac,
259  RE_CA_UART_MAC_BYTES);
260  written += RE_CA_UART_MAC_BYTES;
261  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
262  memcpy (buffer + written,
263  payload->params.adv.adv,
264  payload->params.adv.adv_len);
265  written += payload->params.adv.adv_len;
266  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
267  buffer[written++] = i8tou8 (payload->params.adv.rssi_db);
268  buffer[written++] = RE_CA_UART_FIELD_DELIMITER;
269  buffer[written++] = RE_CA_UART_ETX;
270  *buf_len = written;
271  }
272 
273  return err_code;
274 }
